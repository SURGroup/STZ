\[
\sC(\sbar,T) = \left\{
\begin{array}{ll}
  e^{-\Delta/k_B T} \cosh (\Omega \epsilon_0 \sbar/k_B T) & \qquad \text{for $\Omega \epsilon_0 \sbar < \Delta$} \\
  \Omega \epsilon_0 \sbar / 2\Delta & \qquad \text{for $\Omega \epsilon_0 \sbar \ge \Delta$}
\end{array}
\right.
